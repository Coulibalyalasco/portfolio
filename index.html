CommandeDaoImpl.java — Transaction validation 
// Validation commande : mise à jour atomique des stocks
public boolean changerStatutValidation(String codeCommande, boolean valider, 
                                          AtomicReference<String> errorMsg) {
    cn.setAutoCommit(false);
    
    if (valider) {
        // NON VALIDÉ → VALIDÉ : Ajout au stock et au magasin
        if (appliquerStocksCommande(cn, codeCommande, "ENTREE", errorMsg) &&
            ajouterArticlesMagasin(cn, codeCommande, codeMagasin, errorMsg)) {
            cn.commit();   // ✅ Succès atomique
            return true;
        }
    }
    cn.rollback();   // ❌ Annulation complète en cas d'erreur
    return false;
}
